CREATE TABLE Persona (
	PersKods nchar (12) PRIMARY KEY,
	Vards nvarchar (30) NOT NULL ,
	Uzvards nvarchar (30) NOT NULL ,
	Adrese nvarchar (100) NULL ,
	Telefons nchar (20) NULL ,
	DzimDat datetime2 NOT NULL
)

CREATE TABLE Amats (
	ID int IDENTITY(1,1) PRIMARY KEY,
	VeicejaPK nchar (12) FOREIGN KEY REFERENCES Persona,
	AmataNos nvarchar (30) NOT NULL,
	Apraksts nvarchar (100) NULL
)
